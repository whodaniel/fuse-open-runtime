**Key Methods**:
```typescript

- getAvailableServers()           // Get all servers
- getServersByTags(tags)          // Filter by tags
- getServersByCapabilities(caps)  // Filter by capabilities
```

## 📊 Data Structures Implemented

### 1. MCPServerConfig
```typescript
{
  command: string;              // e.g., "node"
  args: string[];               // e.g., ["--loader", "tsx/esm", "..."]
  env?: Record<string, string>; // Environment variables
  description?: string;         // Human-readable description
  managed_by?: string;          // Managing system
  priority?: number;            // 1-12, higher = more important
  tags?: string[];              // ["api", "automation", ...]
  capabilities?: string[];      // Feature list
}
```

### 2. UserMCPPreferences
```typescript
{
  userId: string;
  enabledServers: string[];           // IDs of enabled servers
  disabledServers: string[];          // Explicitly disabled
  defaultServer?: string;             // Default server ID
  customServers: Record<string, MCPServerConfig>;
  serverOverrides: Record<string, Partial<MCPServerConfig>>;
  lastModified: string;               // ISO timestamp
}
```

### 3. AgentMCPConfig
```typescript
{
  agentId: string;
  agentName: string;
  enabledServers: string[];           // Servers for this agent
  requiredCapabilities: string[];     // Must-have features
  priorityServers: string[];          // Preferred servers
  customTools: string[];              // Agent-specific tools
  toolRestrictions?: {
    allowedTools?: string[];          // Whitelist
    deniedTools?: string[];           // Blacklist
  };
  lastModified: string;
}
```

### 4. WorkflowMCPConfig
```typescript
{
  workflowId: string;
  workflowName: string;
  requiredServers: string[];          // Must be available
  optionalServers: string[];          // Nice to have
  toolMappings: Record<string, string>; // tool → server
  stepMCPConfig: Record<string, string[]>; // step → servers
  lastModified: string;
}
```
