## 📋 Key Specification Highlights

### Breaking Changes in v0.3.0

1. **Well-Known URI Changed**
   - Old: `/.well-known/agent.json`
   - New: `/.well-known/agent-card.json`

2. **mTLS Support Added**
   - New security scheme type: `mutualTLS`
   - Supports client certificate authentication

3. **OAuth2 Metadata URL**
   - New optional field: `oauth2MetadataUrl`
   - Points to RFC 8414 authorization server metadata

4. **Skills Security**
   - Skills can now specify their own security requirements
   - Overrides agent-level security for specific capabilities

5. **AgentCard Signatures**
   - New optional field: `signatures`
   - JWS signatures for agent card verification

6. **Transport Announcement**
   - Multiple transport support via `additionalInterfaces`
   - Prefer JSONRPC, but support gRPC and HTTP+JSON

    // Retrieve task from storage
  }

    // Cancel task, update state
  }

  // Task Management
  private createTask(message: Message): Task {
    const taskId = uuidv4();
    const contextId = message.contextId || uuidv4();

    const task: Task = {
      id: taskId,
      contextId,
      status: {
        state: TaskState.Submitted,
        timestamp: new Date().toISOString(),
      },
      history: [message],
      artifacts: [],
      kind: 'task',
    };

    this.tasks.set(taskId, task);
    return task;
  }

  private updateTaskStatus(
    taskId: string,
    state: TaskState,
    message?: Message
  ) {
    const task = this.tasks.get(taskId);
    if (task) {
      task.status = {
        state,
        message,
        timestamp: new Date().toISOString(),
      };
      this.tasks.set(taskId, task);
    }
  }

  // AgentCard
  getAgentCard(): AgentCard {
    return {
      protocolVersion: A2A_PROTOCOL_VERSION,
      name: 'The New Fuse Agent',
      description: 'AI agent for task automation',
      url: 'https://api.thenewfuse.com/a2a/v1',
      preferredTransport: TransportProtocol.JSONRPC,
      version: '0.3.0',
      capabilities: {
        streaming: true,
        pushNotifications: true,
      },
      defaultInputModes: ['text/plain', 'application/json'],
      defaultOutputModes: ['text/plain', 'application/json'],
      skills: [
        // Define skills
      ],
      securitySchemes: {
        bearer: {
          type: 'http',
          scheme: 'bearer',
          bearerFormat: 'JWT',
        },
      },
      security: [{ bearer: [] }],
    };
  }
}
